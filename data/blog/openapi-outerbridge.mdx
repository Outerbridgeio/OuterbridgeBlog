import { useState } from 'react'
import { templates } from '@/data/template.js'

export const MyDynamicButton = ({buttonText, copyTemplate}) => {
  const [count, setCount] = useState(0)
  const [copied, setCopied] = useState(false)

const onCopy = () => {
setCopied(true)
navigator.clipboard.writeText(JSON.stringify(templates[copyTemplate]))
setTimeout(() => {
setCopied(false)
}, 5000)
}

return <button style={{ height: 50 }}
className={`w-full rounded-md bg-primary-500 py-2 px-4 font-medium text-white hover:bg-primary-700 focus:outline-none focus:ring-2 focus:ring-primary-600 focus:ring-offset-2 dark:ring-offset-black dark:hover:bg-primary-400 sm:py-0`} onClick={onCopy}>{copied ? 'Workflow Copied âœ…' : buttonText } </button>
}

## OpenAI + Outerbridge

![image](https://outerbridgepage.s3.eu-west-1.amazonaws.com/OpenAIXOuterbridge.png)

[OpenAI](https://openai.com/) APIs are now available on Outerbridge! You can now use ChatGPT, Image generation and Text completion powered by OpenAI to build automated workflows such as writing posts, generate content and etc.
To get started, we have prepared 2 workflows:

### ChatGPT to Discord

Write a prompt and have ChatGPT response posted in Discord channel

1. **OpenAI Node**: Provide OpenAI [API key](https://platform.openai.com/account/api-keys) and a
text prompt, then have ChatGPT returning the response

2. **Discord Node**: Get the response from ChatGPT, and post the message to channel

### Create NFT with AI generated image

Write an image description for AI to generate, and automatically create NFT from the image

text prompt, then have OpenAI (Dall-E) returning the generated image in raw format (base64){' '}

2. **Pinata Node**: Since we need to store the image somewhere, we will use Pinata to store in on IPFS

3. **ThirdWeb Node**: Using ThirdWeb and the stored image path, create an NFT
